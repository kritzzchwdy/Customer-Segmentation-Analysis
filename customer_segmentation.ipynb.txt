   "source": ["## Step 5: Elbow Method — Find Optimal K"]
  },
  {
   "cell_type": "code",
   "execution_count": null,
   "metadata": {},
   "outputs": [],
   "source": [
    "X = df_clean[['Annual_Income', 'Spending_Score']]\n",
    "\n",
    "inertia = []\n",
    "for k in range(1, 11):\n",
    "    km = KMeans(n_clusters=k, random_state=42, n_init=10)\n",
    "    km.fit(X)\n",
    "    inertia.append(km.inertia_)\n",
    "\n",
    "plt.figure(figsize=(8, 5))\n",
    "plt.plot(range(1, 11), inertia, marker='o', color='purple', linewidth=2)\n",
    "plt.title('Elbow Method — Finding Optimal K')\n",
    "plt.xlabel('Number of Clusters (K)')\n",
    "plt.ylabel('Inertia (WCSS)')\n",
    "plt.grid(alpha=0.3)\n",
    "plt.savefig('elbow_method.png', dpi=150)\n",
    "plt.show()\n",
    "print('Optimal K = 5 based on elbow bend')"
   ]
